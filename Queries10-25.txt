//Number of Trainings Requested, for a given SPOC Name
  SELECT COUNT(TRF.TRF_ID)
  FROM TRAINING_REQUEST_FORM TRF
  JOIN VERTICAL_TYPE VT
  ON TRF.VER_ID = VT.VER_ID
  JOIN LD_TEAM LT
  ON VT.VER_ID = LT.LD_VER_REP
  WHERE LT.LD_USER_NAME = 'CLINTON EXODIA'
	AND TRF.TRF_DATE_REQUESTED > to_date('30-09-2018', 'dd-MM-yyyy')
	AND TRF.TRF_DATE_REQUESTED < to_date('01-11-2018', 'dd-MM-yyyy');

//Number of Completed Trainings, for a given SPOC Name
SELECT COUNT(CT.TRF_ID)
  FROM CONFIRMED_TRAINING CT
  JOIN TRAINING_REQUEST_FORM TRF
  ON CT.TRF_ID = TRF.TRF_ID
  JOIN VERTICAL_TYPE VT
  ON TRF.VER_ID = VT.VER_ID
  JOIN LD_TEAM LT
  ON VT.VER_ID = LT.LD_VER_REP
  WHERE LT.LD_USER_NAME = 'CLINTON EXODIA'
  AND CT.CT_PROPOSED_END_DATE < sysdate
  AND CT.CT_PROPOSED_END_DATE > to_date('30-09-2018', 'dd-MM-yyyy')
  AND CT.CT_PROPOSED_END_DATE < to_date('01-11-2018', 'dd-MM-yyyy');

//Number of Completed Trainings for given SPOC, grouped by I/D/V, for given SPOC
//For a given month

SELECT TS.TS_DESC, COUNT(CT.TRF_ID)
FROM CONFIRMED_TRAINING CT
JOIN VERTICAL_TYPE VT
ON CT.VER_ID = VT.VER_ID
JOIN LD_TEAM LT
ON VT.VER_ID = LT.LD_VER_REP
JOIN TRAINING_SOURCE TS
ON TS.TS_ID = CT.CT_TRAINING_SOURCE
WHERE LT.LD_USER_NAME = 'CLINTON EXODIA'
AND EXTRACT (MONTH FROM CT.CT_PROPOSED_END_DATE) IN (10)
GROUP BY TS.TS_DESC;
