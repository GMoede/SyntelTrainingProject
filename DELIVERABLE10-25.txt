PRAVEEN QUERIES
//Get all executives

SELECT LT.LD_USER_NAME
FROM LD_TEAM LT
JOIN USER_ROLE UR
ON LT.UR_ID = UR.UR_ID
WHERE UR_NAME = 'EXECUTOR';

//GET STATUS NAME FROM IPT
SELECT OS.OS_NAME
FROM OVERALL_STATUS OS
JOIN IN_PROGRESS_TRAINING IP
ON OS.OS_ID = IP.OS_ID
WHERE
IP.TRF_ID = '***insert_trf_id_here***';

//GET TRF INFO FROM EMAIL
SELECT TRF.TRF_ID, TRF.TRF_TECHNOLOGY, TRF.TRF_DATE_REQUESTED
  FROM TRAINING_REQUEST_FORM TRF
  JOIN VERTICAL_TYPE VT
  ON TRF.VER_ID = VT.VER_ID
  JOIN LD_TEAM LT
  ON VT.VER_ID = LT.LD_VER_REP
  WHERE LT.LD_USER_EMAIL = 'CLINTON_EXODIA@SYNTELINC.COM'
